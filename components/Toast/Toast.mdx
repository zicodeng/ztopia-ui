### Update Toast

    <Button

    </Button>
    <Button
      onClick={() =>
        updateToast('update-demo-toast', {
          containerId: 'update-demo',

      Update Toast
    </Button>
  </Container>

## API

### makeToast, updateToast, dismissToast & dismissAllToasts

```ts
type ToastId = number | string;

const makeToast: (content: ReactNode, options: ToastOptions) => ToastId;

const updateToast: (toastId: ToastId, options: ToastOptions) => void;

const dismissToast: (toastId: ToastId) => void;

const dismissAllToasts: () => void;
```
