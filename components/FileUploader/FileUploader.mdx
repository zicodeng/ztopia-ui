# FileUploader

Dependency: [react-dropzone](https://github.com/react-dropzone/react-dropzone)

## Demo

### Multiple Files

<Playground>
  <FileUploader
    isMulti
    onDropAccepted={(files, e) => {
      console.log('FileUploader', 'onDropAccepted', { files, e });
    }}
  />
</Playground>

### Min & Max FileUploader Size

<Playground>
  <FileUploader
    minSize={1}
    maxSize={100}
    onDropAccepted={(files, e) => {
      console.log('FileUploader', 'onDropAccepted', { files, e });
    }}
    onDropRejected={(files, e) => {
      window.alert('File rejected, size not in range');
      console.log('FileUploader', 'onDropRejected', { files, e });
    }}
  />
</Playground>
