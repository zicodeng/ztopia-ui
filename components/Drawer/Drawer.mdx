### Hide Mask & Custom Close Handler

<Playground>
  <DrawerDemo>
    {({ isOpen, openDrawer, closeDrawer }) => (
      <>
        <Button onClick={openDrawer}>Open</Button>
        <Drawer
          isMaskShown={false}
          isOpen={isOpen}
          onRequestClose={closeDrawer}
          handler={
            <Button
              style={{
                position: 'absolute',
                top: 0,
                zIndex: 1,
              }}
              onClick={closeDrawer}
            >
              Close Drawer
            </Button>
          }
        >
          <section style={{ padding: '20px 40px' }}>
            <h1>Hello, World!</h1>
            <p>{faker.lorem.paragraphs(1)}</p>
