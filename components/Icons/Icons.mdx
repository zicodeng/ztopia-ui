### IconBell

<Playground>
  <Container>
    <IconBellDemo>
      {({ count, setCount }) => (
        <>
          <IconBell isAnimated count={count} size="large" />
          <Button
            onClick={() => {
              setCount(++count);
            }}
          >
            Increase Count
          </Button>
          <Button
            onClick={() => {
              if (count > 0) {
                setCount(--count);
              }
            }}
          >
            Decrease Count
          </Button>
        </>
      )}
    </IconBellDemo>
  </Container>
</Playground>
