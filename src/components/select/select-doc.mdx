## Basic Select

### Select Mode - Single Selection
<Playground className={playgroundStyle}>
    <Select
        placeholder="What Is Your Favorite Fruit?"
        options={[
            {
                key: 'apple',
                label: 'Apple',
                value: 'apple'
            },
            {
                key: 'banana',
                label: 'Banana',
                value: 'banana'
            },
            {
                key: 'orange',
                label: 'Orange',
                value: 'orange'
            },
            {
                key: 'avocado',
                label: 'Avocado',
                value: 'avocado'
            },
            {
                key: 'pineapple',
                label: 'Pineapple',
                value: 'pineapple'
            },
            {
                key: 'mango',
                label: 'Mango',
                value: 'mango'
            },
                        {
                key: 'watermelon',
                label: 'Watermelon',
                value: 'watermelon'
            },
        ]}
        onSelect={(selectedOption) => console.log('SELECTED OPTION:', selectedOption)}
        onDeselect={(deselectedOption) => console.log('DESELECTED OPTION:', deselectedOption)}
        onSelectionChange={(data) => console.log('CURRENTLY SELECTED:', data)}
    />
</Playground>
